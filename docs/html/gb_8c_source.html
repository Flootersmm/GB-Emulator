544void step(GB *vm) {
545 FILE *logfile = fopen("cpu_log.txt", "a");
546 if (logfile) {
547 fprintf(logfile,
548 "A:%02X F:%02X B:%02X C:%02X D:%02X E:%02X H:%02X L:%02X SP:%04X "
549 "PC:%04X PCMEM:%02X,%02X,%02X,%02X\n",
550 vm->r.a, vm->r.f, vm->r.b, vm->r.c, vm->r.d, vm->r.e, vm->r.h,
551 vm->r.l, vm->r.sp, vm->r.pc, read_u8(vm, vm->r.pc),
552 read_u8(vm, vm->r.pc + 1), read_u8(vm, vm->r.pc + 2),
553 read_u8(vm, vm->r.pc + 3));
554 fclose(logfile);
555 }
556
557 if (vm->flag.halt) {
558 return;
559 }
560
561 u8 opcode = vm->mem.data[vm->r.pc++];
562 const OPS *instr = &ops[opcode];
563 int cycles = op_ticks[opcode];
564
565 // printf("op: %02X, pushed: %02X\n", opcode, vm->mem.data[0xdffd - 4]);
566
567 switch (instr->type) {
568 case NO_OP:
569 if (instr->func.func_no_op) {
570 instr->func.func_no_op(vm);
571 }
572 break;
573 case U8_OP:
574 if (instr->func.func_u8_op) {
575 u8 operand = read_u8(vm, vm->r.pc);
576 instr->func.func_u8_op(vm, operand);
577 }
578 break;
579 case U16_OP:
580 if (instr->func.func_u16_op) {
581 u16 operand = read_u16(vm, vm->r.pc);
582 instr->func.func_u16_op(vm, operand);
583 }
584 break;
585 }
586
587 if (opcode != 0xC3) {
588 vm->r.pc += instr->length;
589 }
590
591 vm->cycles += cycles;
592
593 update_timers(vm, cycles);
594 update_graphics(vm, cycles);
595
596 if (vm->r.pc == 0x282a) {
597 printf("Printing Tetris tileset to tile0.bin\n");
598 FILE *f = fopen("tile0.bin", "wb");
599 if (f) {
600 fwrite(&(vm->mem.data[0x8000]), 16, 1, f);
601 fclose(f);
602 }
603 }
604
605 // DI and EI have a delay.
606 if (vm->flag.interrupt_disable_pending) {
607 vm->flag.interrupt_master_enable = false;
608 vm->flag.interrupt_disable_pending = false;
609 }
610 if (vm->flag.interrupt_enable_pending) {
611 vm->flag.interrupt_master_enable = true;
612 vm->flag.interrupt_enable_pending = false;
613 }
614
615 do_interrupts(vm);
616}

620

622 return;
623 }

639 }

643void set_lcd_status(GB *vm) {
644 u8 status = read_u8(vm, 0xFF41);
645 if (!is_lcd_enabled(vm)) {
646 vm->scanline_counter = 456;
647 vm->mem.data[0xFF44] = 0;
648 status &= 252;
649 status |= 1;
650 write_u8(vm, 0xFF41, status);
651 return;
652 }
653
654 u8 current_line = read_u8(vm, 0xFF44);
655 u8 current_mode = status & 0x3;
656 u8 mode = 0;
657 bool req_int = false;
658
659 if (current_line >= 144) {
660 mode = 1;
661 status |= 1;
662 status &= ~2;
663 req_int = status & (1 << 4);
664 } else {
665 int mode2_bounds = 456 - 80;
666 int mode3_bounds = mode2_bounds - 172;
667
668 if (vm->scanline_counter >= mode2_bounds) {
669 mode = 2;
670 status |= 2;
671 status &= ~1;
672 req_int = status & (1 << 5);
673 } else if (vm->scanline_counter >= mode3_bounds) {
674 mode = 3;
675 status |= 3;
676 } else {
677 mode = 0;
678 status &= ~3;
679 req_int = status & (1 << 3);
680 }
681 }
682
683 if (req_int && (mode != current_mode)) {
684 request_interrupt(vm, 1);
685 }
686
687 if (current_line == read_u8(vm, 0xFF45)) {
688 status |= 4;
689 if (status & (1 << 6)) {
690 request_interrupt(vm, 1);
691 }
692 } else {
693 status &= ~4;
694 }
695
696 write_u8(vm, 0xFF41, status);
697}

797void do_dma_transfer(GB *vm, u8 data) {

901void service_interrupt(GB *vm, u8 interrupt) {

913 case 1:

916 case 2:
