410void draw_disassembly_window(GB *vm) {
411ImGui::Begin("Disassembly Window", NULL, ImGuiWindowFlags_AlwaysAutoResize | ImGuiWindowFlags_NoScrollbar | ImGuiWindowFlags_NoScrollWithMouse | ImGuiWindowFlags_NoResize);
412
413 // Display around the current PC
414 u16 start_address = vm->r.pc > 0x10 ? vm->r.pc - 0x10 : 0;
415 u16 end_address = start_address + 0x20;
416
417 for (u16 address = start_address; address < end_address; ) {
418 u8 opcode = vm->mem.data[address];
419 const char *op_str = ops[opcode].debug_str;
420 int operand_length = ops[opcode].length;
421
422 // Print address
423 ImGui::Text("%04X: ", address);
424
425 // Highlight the current PC
426 if (address == vm->r.pc) {
427 ImGui::SameLine();
428 ImGui::TextColored(ImVec4(1.0f, 0.0f, 0.0f, 1.0f), "%s ", op_str);
429 } else {
430 ImGui::SameLine();
431 ImGui::Text("%s ", op_str);
432 }
433
434 // Print operands
435 for (int i = 1; i <= operand_length; i++) {
436 ImGui::SameLine();
437 ImGui::Text("%02X ", vm->mem.data[address + i]);
438 }
439
440 address += 1 + operand_length;
441 }
442
443 ImGui::End();
444}
